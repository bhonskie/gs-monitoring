        async function bootstrapFirebase() {
            const appId = typeof __app_id !== 'undefined' ? __app_id : 'gs-attendance-app-persisted';
            let firebaseConfig = null;
            try {
                if (typeof __firebase_config !== 'undefined' && __firebase_config) {
                    firebaseConfig = JSON.parse(__firebase_config);
                }
            } catch (err) {
                console.error("Firebase config parsing error:", err);
            }

            if (!firebaseConfig) {
                console.warn("Firebase config not available. Standing by with persistent in-memory operations.");
                setupLiveInMemorySampleData();
                populateEmployeeDropdown();
                renderAdminDashboard();
                return;
            }

            try {
                const app = initializeApp(firebaseConfig);
                db = getFirestore(app);
                auth = getAuth(app);

                let user = null;
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    const cred = await signInWithCustomToken(auth, __initial_auth_token);
                    user = cred.user;
                } else {
                    const cred = await signInAnonymously(auth);
                    user = cred.user;
                }

                if (user) {
                    isUsingFirebase = true;
                    setupFirestoreSync(appId);
                } else {
                    throw new Error("Anonymous auth returned null user.");
                }
            } catch (err) {
                console.error("Failed to connect to Firebase. Using offline local fallback state:", err);
                setupLiveInMemorySampleData();
                populateEmployeeDropdown();
                renderAdminDashboard();
            }
        }

        function setupFirestoreSync(appId) {
            // STRICT RULE 3 Guard checking
            if (!auth.currentUser) return;

            // 1. Sync System Administrators
            const adminsCol = collection(db, 'artifacts', appId, 'public', 'data', 'admins');
            onSnapshot(adminsCol, (snapshot) => {
                let tempAdmins = [];
                snapshot.forEach(doc => {
                    tempAdmins.push(doc.data());
                });
                if (!tempAdmins.some(a => a.user.toLowerCase() === 'admin')) {
                    tempAdmins.push({ user: 'admin', pass: '123' });
                }
                adminUsers = tempAdmins;
                renderAdminDashboard();
            }, (error) => {
                console.error("Admins collection sync failure:", error);
            });

            // 2. Sync Employee Profiles
            const employeesCol = collection(db, 'artifacts', appId, 'public', 'data', 'employees');
            onSnapshot(employeesCol, (snapshot) => {
                let tempEmployees = [];
                snapshot.forEach(doc => {
                    tempEmployees.push(doc.data());
                });
                if (tempEmployees.length === 0) {
                    seedDefaultEmployees(appId);
                } else {
                    rawMasterDataset = tempEmployees;
                    populateEmployeeDropdown();
                    renderAdminDashboard();
                    renderComplianceAudit();
                }
            }, (error) => {
                console.error("Employees collection sync failure:", error);
            });

            // 3. Sync Active Logs
            const logsCol = collection(db, 'artifacts', appId, 'public', 'data', 'logs');
            onSnapshot(logsCol, (snapshot) => {
                let tempLogs = [];
                snapshot.forEach(doc => {
                    const data = doc.data();
                    tempLogs.push({
                        ...data,
                        loginRaw: data.loginRawTime ? new Date(data.loginRawTime) : null,
                        logoutRaw: data.logoutRawTime ? new Date(data.logoutRawTime) : null,
                        breakStartRaw: data.breakStartRawTime ? new Date(data.breakStartRawTime) : null,
                        breakEndRaw: data.breakEndRawTime ? new Date(data.breakEndRawTime) : null
                    });
                });
                if (tempLogs.length === 0 && rawMasterDataset.length > 0) {
                    seedDefaultLogs(appId);
                } else {
                    attendanceLogs = tempLogs;
                    renderAdminDashboard();
                    renderEmployeeDashboard();
                    renderComplianceAudit();
                }
            }, (error) => {
                console.error("Logs collection sync failure:", error);
            });

            // 4. Sync Roster Remarks
            const remarksCol = collection(db, 'artifacts', appId, 'public', 'data', 'remarks');
            onSnapshot(remarksCol, (snapshot) => {
                let tempRemarks = {};
                snapshot.forEach(doc => {
                    try {
                        tempRemarks[doc.id] = JSON.parse(doc.data().remarksMap || '{}');

                        tempRemarks[doc.id] = doc.data().remarksMap || {};

        function getMondayOfDate(d) {
            const date = new Date(d);
            const day = date.getDay();
            const diff = date.getDate() - day + (day === 0 ? -6 : 1); 
            return new Date(date.setDate(diff));
        }

        function auditEmployeeAttendance(empName, anchorDateStr) {
            const anchor = new Date(anchorDateStr);
            const mon = getMondayOfDate(anchor);
            const weekdays = [];
            
            for (let i = 0; i < 7; i++) {
                const d = new Date(mon);
                d.setDate(mon.getDate() + i);
                weekdays.push(d.toISOString().split('T')[0]);
            }

            let totalAbsences = 0;
            let maxConsecutive = 0;
            let currentConsecutive = 0;
            const dailyStatus = [];

            const todayStr = new Date().toISOString().split('T')[0];
            const empMeta = rawMasterDataset.find(e => e.name === empName);
            const daysOff = empMeta && empMeta.daysOff ? empMeta.daysOff : ["Saturday", "Sunday"];

            weekdays.forEach((dateStr) => {
                const hasLog = attendanceLogs.some(l => l.name === empName && l.date === dateStr && l.loginRaw);
                const remark = rosterRemarksDb[dateStr]?.[empName] || "";

                let status = "untracked";
                
                const dayDate = new Date(dateStr);
                const dayName = dayDate.toLocaleDateString('en-US', { weekday: 'long' });

                if (daysOff.includes(dayName)) {
                    status = "Rest";
                } else if (hasLog) {
                    status = "Present";
                } else if (remark === "Sick" || remark === "on Leave") {
                    status = "Excused";
                } else if (remark === "Absent") {
                    status = "Absent";
                } else {
                    if (dateStr <= todayStr) {
                        status = "Absent";
                    } else {
                        status = "Pending";
                    }
                }

                dailyStatus.push({ date: dateStr, status: status, remark: remark });

                if (status === "Absent") {
                    totalAbsences++;
                    currentConsecutive++;
                    if (currentConsecutive > maxConsecutive) {
                        maxConsecutive = currentConsecutive;
                    }
                } else if (status === "Present" || status === "Excused") {
                    currentConsecutive = 0;
                }
            });

            let calculatedStatus = "Active";
            
            if (lockedUsersDb[empName] === "LOCKED" || totalAbsences > 3) {
                calculatedStatus = "Lock the User";
            } 
            else if (maxConsecutive >= 3) {
                calculatedStatus = "Verbal/Written Warning";
            }

            return {
                weekdays,
                dailyStatus,
                totalAbsences,
                maxConsecutive,
                status: calculatedStatus
            };
        }

        function calculateLateness(loginTime, assignedDutyTimeStr) {
            if (!assignedDutyTimeStr) return { isLate: false, latenessText: 'On Time', latenessMins: 0, dutyDate: null };
            let cleanStr = assignedDutyTimeStr.trim().toLowerCase().replace(/(\d+)(am|pm)/, "$1 $2");
            const parts = cleanStr.split(' ');
            if (parts.length < 2) return { isLate: false, latenessText: 'On Time', latenessMins: 0, dutyDate: null };

            const timeParts = parts[0].split(':');
            let dutyHour = parseInt(timeParts[0]);
            const dutyMinute = parseInt(timeParts[1] || '0');
            const ampm = parts[1];

            if (isNaN(dutyHour)) return { isLate: false, latenessText: 'On Time', latenessMins: 0, dutyDate: null };
            if (ampm === 'pm' && dutyHour < 12) dutyHour += 12;
            if (ampm === 'am' && dutyHour === 12) dutyHour = 0;

            const dutyDate = new Date(loginTime);
            dutyDate.setHours(dutyHour, dutyMinute, 0, 0);

            const diffMins = Math.floor((loginTime - dutyDate) / (1000 * 60));
            if (diffMins > 5) {
                return { isLate: true, latenessText: `LATE by ${diffMins} mins`, latenessMins: diffMins, dutyDate: dutyDate };
            }
            return { isLate: false, latenessText: 'On Time', latenessMins: 0, dutyDate: dutyDate };
        }

            if(actionType === 'In') {
                if(logIndex > -1 && attendanceLogs[logIndex].loginRaw) {
                    return triggerCustomAlert(`A log-in record already exists for ${empName} on ${chosenDateStr}.`);
                }

                const expectedStartStr = document.getElementById('loginShiftStart').value;
                const lateInfo = calculateLateness(targetLogTime, expectedStartStr);

                const newLog = {
                    name: empName,
                    level: empMeta ? empMeta.level : 'Level A',
                    date: chosenDateStr,
                    expectedDuty: expectedStartStr,
                    loginRaw: targetLogTime,
                    logoutRaw: null,
                    login: targetLogTime.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }),
                    logout: '',
                    isLate: lateInfo.isLate,
                    latenessMins: lateInfo.latenessMins || 0,
                    lateRemarks: lateInfo.latenessText,
                    regHrs: '-', ot: '-', ut: '-',
                    breakStartRaw: null,
                    breakEndRaw: null,
                    breakStart: '',
                    breakEnd: '',
                    breakDuration: 0,
                    isOverbreak: false,
                    overbreakMins: 0
                };
                
                if(logIndex > -1) attendanceLogs[logIndex] = newLog;
                else attendanceLogs.push(newLog);

                await saveLogToDb(newLog);
                
                if (lateInfo.isLate) {
                    if (!rosterRemarksDb[chosenDateStr]) {
                        rosterRemarksDb[chosenDateStr] = {};
                    }
                    rosterRemarksDb[chosenDateStr][empName] = "Late";
                    await saveRemarksToDb(chosenDateStr, rosterRemarksDb[chosenDateStr]);
                    showNotificationToast(`Auto-updated "Late" remark for ${empName}`);
                }

                showNotificationToast(`Session Log-In Synchronized for ${empName}`);
                triggerCustomAlert(`Log In registered successfully for ${chosenDateStr} at ${newLog.login}.\nStatus: ${lateInfo.latenessText}`);
            } else {
                if(logIndex === -1 || !attendanceLogs[logIndex].loginRaw) {
                    return triggerCustomAlert(`Cannot locate a corresponding active Log-In session for ${chosenDateStr}.`);
                }
                
                let currentLog = attendanceLogs[logIndex];
                currentLog.logoutRaw = targetLogTime;
                currentLog.logout = targetLogTime.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
                
                const expectedStartStr = currentLog.expectedDuty;
                const lateInfo = calculateLateness(currentLog.loginRaw, expectedStartStr);
                const dutyDate = lateInfo.dutyDate;

                let regHrsValue = 0;
                let otHrsValue = 0;
                let utHrsValue = 0;

                if (dutyDate) {
                    const standardShiftEnd = new Date(dutyDate.getTime() + 8 * 60 * 60 * 1000);
                    const latenessHours = lateInfo.isLate ? (lateInfo.latenessMins / 60) : 0;
                    
                    const startForReg = currentLog.loginRaw > dutyDate ? currentLog.loginRaw : dutyDate;
                    const endForReg = currentLog.logoutRaw < standardShiftEnd ? currentLog.logoutRaw : standardShiftEnd;

                    regHrsValue = (endForReg - startForReg) / (1000 * 60 * 60);
                    if (regHrsValue < 0) regHrsValue = 0;
                    
                    const maxRegHoursAllowed = 8 - latenessHours;
                    if (regHrsValue > maxRegHoursAllowed) regHrsValue = maxRegHoursAllowed;

                    if (currentLog.logoutRaw > standardShiftEnd) {
                        otHrsValue = (currentLog.logoutRaw - standardShiftEnd) / (1000 * 60 * 60);
                    } else {
                        otHrsValue = 0;
                    }

                    utHrsValue = 8 - regHrsValue;
                    if (utHrsValue < 0) utHrsValue = 0;
                } else {
                    let diffHrs = (currentLog.logoutRaw - currentLog.loginRaw) / (1000 * 60 * 60);
                    if (diffHrs < 0) diffHrs = 8;
                    
                    if (diffHrs >= 8) {
                        regHrsValue = 8;
                        otHrsValue = diffHrs - 8;
                        utHrsValue = 0;
                    } else {
                        regHrsValue = diffHrs;
                        otHrsValue = 0;
                        utHrsValue = 8 - diffHrs;
                    }
                }

                currentLog.regHrs = `${regHrsValue.toFixed(1)} hrs`;
                currentLog.ot = `${otHrsValue.toFixed(1)} hrs`;
                currentLog.ut = `${utHrsValue.toFixed(1)} hrs`;

                await saveLogToDb(currentLog);

                showNotificationToast(`Session Log-Out Synchronized for ${empName}`);
                triggerCustomAlert(`Log Out recorded successfully for ${chosenDateStr} at ${currentLog.logout}\nRegular Hours: ${currentLog.regHrs}\nCalculated Undertime: ${currentLog.ut}`);
            }

            handleTerminalUserChange();
            renderAdminDashboard();
        }

        async function handleBreakAction(actionType) {
            const empName = document.getElementById('empSelect').value;
            if(!empName) return triggerCustomAlert('Please choose an active profile identity first.');

            const chosenDateStr = document.getElementById('employeeAttendanceDate').value;
            if(!chosenDateStr) return triggerCustomAlert('Please pick a valid log date.');

            let logIndex = attendanceLogs.findIndex(l => l.name === empName && l.date === chosenDateStr);

            if(logIndex === -1 || !attendanceLogs[logIndex].loginRaw) {
                return triggerCustomAlert(`Active log session required. You must Log-In first before starting a break.`);
            }

            let log = attendanceLogs[logIndex];
            let rightNow = new Date();

            if (actionType === 'Start') {
                if (log.breakStartRaw) {
                    return triggerCustomAlert(`An existing break session was already initiated today at ${log.breakStart}.`);
                }

                log.breakStartRaw = rightNow;
                log.breakStart = rightNow.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
                
                if (!rosterRemarksDb[chosenDateStr]) {
                    rosterRemarksDb[chosenDateStr] = {};
                }
                rosterRemarksDb[chosenDateStr][empName] = "On Break";
                
                await saveLogToDb(log);
                await saveRemarksToDb(chosenDateStr, rosterRemarksDb[chosenDateStr]);

                showNotificationToast(`Break Registered for ${empName}`);
                triggerCustomAlert(`Break started successfully for ${empName} at ${log.breakStart}.\nDashboard Status: On Break.`);
            } 
            else if (actionType === 'End') {
                if (!log.breakStartRaw) {
                    return triggerCustomAlert(`You must initiate a break start first before concluding it.`);
                }
                if (log.breakEndRaw) {
                    return triggerCustomAlert(`This break session was already completed at ${log.breakEnd}.`);
                }

                log.breakEndRaw = rightNow;
                log.breakEnd = rightNow.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });

                const simMode = document.getElementById('breakSimMode').value;
                let calculatedMins = 0;

                if (simMode === 'real') {
                    const msDiff = log.breakEndRaw - log.breakStartRaw;
                    calculatedMins = Math.floor(msDiff / (1000 * 60));
                    if (calculatedMins <= 0) calculatedMins = 1;
                } else {
                    calculatedMins = parseInt(simMode);
                }

                log.breakDuration = calculatedMins;

                if (calculatedMins > 60) {
                    log.isOverbreak = true;
                    log.overbreakMins = calculatedMins - 60;
                    
                    if (!rosterRemarksDb[chosenDateStr]) {
                        rosterRemarksDb[chosenDateStr] = {};
                    }
                    rosterRemarksDb[chosenDateStr][empName] = "Overbreak";
                    
                    await saveLogToDb(log);
                    await saveRemarksToDb(chosenDateStr, rosterRemarksDb[chosenDateStr]);

                    showNotificationToast(`Auto-assigned "Overbreak" status warning.`);
                    triggerCustomAlert(`Break Session Closed! Total Break Duration: ${calculatedMins} minutes.\n🚨 Overbreak registered: Exceeded standard limit by ${log.overbreakMins} minutes.`);
                } else {
                    log.isOverbreak = false;
                    log.overbreakMins = 0;
                    
                    if (rosterRemarksDb[chosenDateStr] && rosterRemarksDb[chosenDateStr][empName] === "On Break") {
                        delete rosterRemarksDb[chosenDateStr][empName];
                    }
                    
                    await saveLogToDb(log);
                    await saveRemarksToDb(chosenDateStr, rosterRemarksDb[chosenDateStr]);

                    showNotificationToast(`Break session concluded on-time.`);
                    triggerCustomAlert(`Break Session Closed! Total Break Duration: ${calculatedMins} minutes. (Completed On-Time within limit)`);
                }
            }

            renderEmployeeDashboard();
            renderAdminDashboard();
        }

        function matchesTimeFilter(logDateStr, anchorDateStr, targetMode) {
            if (!logDateStr) return false;
            if (targetMode === 'day') return logDateStr === anchorDateStr;
            
            const logDate = new Date(logDateStr);
            const anchorDate = new Date(anchorDateStr);
            if (isNaN(logDate.getTime()) || isNaN(anchorDate.getTime())) return false;

            if (targetMode === 'month') {
                return logDate.getFullYear() === anchorDate.getFullYear() && logDate.getMonth() === anchorDate.getMonth();
            }

            if (targetMode === 'week') {
                const currentDayOfWeek = anchorDate.getDay(); 
                const startOfWeek = new Date(anchorDate);
                startOfWeek.setDate(anchorDate.getDate() - currentDayOfWeek);
                startOfWeek.setHours(0,0,0,0);
                const endOfWeek = new Date(startOfWeek);
                endOfWeek.setDate(startOfWeek.getDate() + 6);
                endOfWeek.setHours(23,59,59,999);
                return logDate >= startOfWeek && logDate <= endOfWeek;
            }
            return true;
        }

        function getFilteredAdminLogs(logs, anchorDateStr, targetMode) {
            let filteredLogs = logs.filter(l => matchesTimeFilter(l.date, anchorDateStr, targetMode));
            const statusFilter = document.getElementById('adminStatusFilter').value;
            
            if (statusFilter === 'late') {
                filteredLogs = filteredLogs.filter(log => log.isLate === true);
            } else if (statusFilter === 'ontime') {
                filteredLogs = filteredLogs.filter(log => log.isLate === false);
            } else if (statusFilter === 'onbreak') {
                filteredLogs = filteredLogs.filter(log => log.breakStartRaw && !log.breakEndRaw);
            } else if (statusFilter === 'overbreak') {
                filteredLogs = filteredLogs.filter(log => log.isOverbreak === true);
            } else if (statusFilter === 'undertime') {
                filteredLogs = filteredLogs.filter(log => parseFloat(log.ut) > 0);
            } else if (statusFilter === 'shift12') {
                filteredLogs = filteredLogs.filter(log => log.expectedDuty.toLowerCase() === '12:00 am');
            } else if (statusFilter === 'shift2') {
                filteredLogs = filteredLogs.filter(log => log.expectedDuty.toLowerCase() === '2:00 am');
            } else if (statusFilter === 'shift3') {
                filteredLogs = filteredLogs.filter(log => log.expectedDuty.toLowerCase() === '3:00 am');
            } else if (statusFilter === 'late12') {
                filteredLogs = filteredLogs.filter(log => log.isLate === true && log.expectedDuty.toLowerCase() === '12:00 am');
            } else if (statusFilter === 'late2') {
                filteredLogs = filteredLogs.filter(log => log.isLate === true && log.expectedDuty.toLowerCase() === '2:00 am');
            } else if (statusFilter === 'late3') {
                filteredLogs = filteredLogs.filter(log => log.isLate === true && log.expectedDuty.toLowerCase() === '3:00 am');
            } else if (statusFilter === 'late_both') {
                filteredLogs = filteredLogs.filter(log => log.isLate === true && (log.expectedDuty.toLowerCase() === '12:00 am' || log.expectedDuty.toLowerCase() === '2:00 am'));
            }
            return filteredLogs;
        }

                const calendarDays = [];

                    calendarDays.forEach(dateStr => {
                        const dayDate = new Date(dateStr);
                        const dayName = dayDate.toLocaleDateString('en-US', { weekday: 'long' });

                        // Skip if scheduled rest day off
                        if (daysOff.includes(dayName)) return;

                        totalWorkDays++;

                        const hasLog = attendanceLogs.some(l => l.name === emp.name && l.date === dateStr && l.loginRaw);
                        const remark = rosterRemarksDb[dateStr]?.[emp.name] || "";

                        if (hasLog) {
                            daysPresent++;
                        } else if (remark === "Sick" || remark === "on Leave") {
                            excused++;
                        } else {
                            const todayStr = new Date().toISOString().split('T')[0];
                            if (dateStr <= todayStr || remark === "Absent") {
                                unexcused++;
                            }

                    const complianceScore = totalWorkDays > 0 ? Math.round(((daysPresent + excused) / totalWorkDays) * 100) : 100;

                        ratingText = `<span class="px-2 py-0.5 rounded text-[9px] bg-amber-100 text-amber-900 font-bold">Needs Improvement</span>`;
                    }

                    const tr = document.createElement('tr');
                    tr.className = "hover:bg-slate-50/50 border-b border-slate-100 transition";
                    tr.innerHTML = `
                        <td class="p-3 font-bold text-slate-800">${emp.name}</td>
                        <td class="p-3 text-slate-500 font-semibold">${emp.position} (${emp.dutyTime})</td>
                        <td class="p-3 text-center font-bold text-slate-700">${totalWorkDays}</td>
                        <td class="p-3 text-center font-bold text-emerald-700">${daysPresent}</td>
                        <td class="p-3 text-center font-bold text-rose-600">${unexcused}</td>
                        <td class="p-3 text-center font-bold text-amber-600">${excused}</td>
                        <td class="p-3 text-center font-extrabold text-slate-900">${complianceScore}%</td>
                        <td class="p-3 text-center">${ratingText}</td>
                    `;
                    tbody.appendChild(tr);
                });
            }

            updateLockedBadgeCounter();
        }

        function updateLockedBadgeCounter() {
            const badge = document.getElementById('lockedCountBadge');
            if (!badge) return;
            
            const weekAnchorDate = document.getElementById('complianceWeekAnchor').value || new Date().toISOString().split('T')[0];
            let lockedCount = 0;

            rawMasterDataset.forEach(emp => {
                const audit = auditEmployeeAttendance(emp.name, weekAnchorDate);
                if (audit.status === "Lock the User") {
                    lockedCount++;
                }
            });

            if (lockedCount > 0) {
                badge.textContent = lockedCount;
                badge.classList.remove('hidden');
            } else {
                badge.classList.add('hidden');
            }
        }

        async function handleEmployeeSelfRegister(e) {
            e.preventDefault();
            const dutyTimeVal = document.getElementById('regEmpDutyTime').value.trim();
            const levelVal = document.getElementById('regEmpLevel').value;
            const positionVal = document.getElementById('regEmpPosition').value;
            const nameVal = document.getElementById('regEmpName').value.trim();
            
            if (rawMasterDataset.some(emp => emp.name.toLowerCase() === nameVal.toLowerCase())) {
                return triggerCustomAlert('This profile name is already registered in our active database.');
            }

            if (!dutyTimeVal.toLowerCase().includes('am') && !dutyTimeVal.toLowerCase().includes('pm')) {
                return triggerCustomAlert('Please specify AM or PM on your Shift Start field (e.g. 12:00 AM).');
            }

            const daysOffCheckboxes = document.querySelectorAll('input[name="regEmpDaysOff"]:checked');
            const selectedDaysOff = Array.from(daysOffCheckboxes).map(cb => cb.value);

            if (selectedDaysOff.length === 0) {
                return triggerCustomAlert('Please designate at least one official Rest Day.');
            }

            const newEmployee = {
                name: nameVal, 
                position: positionVal, 
                level: levelVal, 
                dutyTime: dutyTimeVal, 
                status: "ACTIVE",
                daysOff: selectedDaysOff
            };

            rawMasterDataset.push(newEmployee);
            
            // PERSIST TO DATABASE
            await saveEmployeeToDb(newEmployee);

            triggerCustomAlert('Profile saved and registered successfully!');
            showNotificationToast(`Registered profile: ${nameVal}`);
            e.target.reset();
            
            document.querySelectorAll('input[name="regEmpDaysOff"]').forEach(cb => {
                if (cb.value === "Saturday" || cb.value === "Sunday") {
                    cb.checked = true;
                } else {
                    cb.checked = false;
                }
            });

            toggleModal('registerModal', false);
            populateEmployeeDropdown();
            renderAdminDashboard();
        }

        function exportRosterDatabase() {
            const remarksFilterValue = document.getElementById('rosterRemarksFilter').value;
            const selectedRosterDate = document.getElementById('rosterDateAnchor').value || new Date().toISOString().split('T')[0];
            
            let exportPayload = [];

            if (currentRosterFilterTab === 'day') {
                rawMasterDataset.forEach(emp => {
                    const empRemark = rosterRemarksDb[selectedRosterDate]?.[emp.name] || "No Remarks";
                    const empDaysOff = emp.daysOff || ["Saturday", "Sunday"];
                    if (remarksFilterValue === 'none' && empRemark !== "No Remarks") return;
                    if (remarksFilterValue !== 'all' && remarksFilterValue !== 'none' && empRemark !== remarksFilterValue) return;

                    exportPayload.push({
                        "Employee Name": emp.name,
                        "Position": emp.position,
                        "Grade Level": emp.level,
                        "Assigned Shift": emp.dutyTime,
                        "Designated Days Off": empDaysOff.join(', '),
                        "Roster Date": selectedRosterDate,
                        "Remarks": empRemark
                    });
                });
            } else {
                for (let dateKey in rosterRemarksDb) {
                    if (matchesTimeFilter(dateKey, selectedRosterDate, currentRosterFilterTab)) {
                        for (let empName in rosterRemarksDb[dateKey]) {
                            const remarkVal = rosterRemarksDb[dateKey][empName];
                            if (!remarkVal) continue;
                            if (remarksFilterValue !== 'all' && remarksFilterValue !== remarkVal) continue;

                            const empMeta = rawMasterDataset.find(e => e.name === empName);
                            const empDaysOff = empMeta && empMeta.daysOff ? empMeta.daysOff : ["Saturday", "Sunday"];
                            exportPayload.push({
                                "Employee Name": empName,
                                "Position": empMeta ? empMeta.position : "N/A",
                                "Grade Level": empMeta ? empMeta.level : "N/A",
                                "Assigned Shift": empMeta ? empMeta.dutyTime : "N/A",
                                "Designated Days Off": empDaysOff.join(', '),
                                "Roster Date": dateKey,
                                "Remarks": remarkVal
                            });
                        }
                    }
                }
            }

            if (exportPayload.length === 0) {
                return triggerCustomAlert('No records tracked matching the current dashboard selections.');
            }

            const worksheet = XLSX.utils.json_to_sheet(exportPayload);
            const workbook = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(workbook, worksheet, "Remarks History");
            XLSX.writeFile(workbook, `Roster_Remarks_Report_${selectedRosterDate}.xlsx`);
        }
